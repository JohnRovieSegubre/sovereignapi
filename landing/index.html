Five Lines of Code. That's It.

python (sovereign-sdk)
from cdp.x402.fetch import x402Fetch # Pure x402 Mode (Recommended) client = x402Fetch(wallet_signer, base_url="https://api.sovereign-api.com") # No API keys or accounts. Just pay and think. resp = await client.post("/v1/chat/completions", json={ "model": "sovereign/claude-3.7-sonnet", "messages": [{"role": "user", "content": "Analyze this loop."}] })
